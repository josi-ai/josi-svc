**Goal:** Decouple from Clerk, add Redis session caching for instant auth resolution, add usage tracking, and make auth provider switching seamless.

**Architecture:** Rename `clerk_id` → `auth_provider_id` in DB. Add Redis-backed session cache so auth middleware resolves users in sub-millisecond (cache hit) instead of DB/JWT parsing. ## Context

### Current State
- User model has `clerk_id` field (tightly coupled to Clerk)
- JWT validation hardcodes Clerk's JWKS URL
- Clerk publicMetadata is used to inject custom claims into JWT
- Redis exists in codebase (`src/josi/core/cache.py`) but is optional and not in docker-compose
- Auth middleware resolves users from JWT claims or DB on every request
- No usage tracking exists

### Target State
- User model has `auth_provider_id` + `auth_provider` (provider-agnostic)
- JWT validation is pluggable via provider adapters
- Redis caches user sessions — sub-millisecond auth resolution
- Cache invalidation on logout, subscription change, role change
- Monthly usage counters in Redis + DB
- Users can view their usage via `GET /api/v1/me/usage`

### Layering Convention (atlas-svc pattern)
```
Controller → Service(current_user=current_user) → Repository(current_user=current_user)
                                                      └→ get_async_session() internally
```
- Controllers: I/O validation only
- Services: business logic, created with `current_user` context
- Repositories: DB access via `get_async_session()` context manager, no injected session

"""Auth provider abstraction — swap providers via config."""
from abc import ABC, abstractmethod
from functools import lru_cache

import structlog

logger = structlog.get_logger()


class AuthProvider(ABC):
    """Interface for JWT validation providers."""

    @abstractmethod
    def validate_jwt(self, token: str) -> dict:
        """Validate a JWT and return decoded claims.

        Must raise fastapi.HTTPException(401) on failure.
        Returns dict with at minimum: sub, email.
        """
        ...

    @abstractmethod
    async def set_user_metadata(self, provider_user_id: str, metadata: dict) -> bool:
        """Set custom metadata on the provider's user record.

        Used to inject custom claims into the provider's JWT template.
        Returns True on success.
        """
        ...

    @property
    @abstractmethod
    def provider_name(self) -> str:
        """Return the provider identifier (e.g. 'clerk', 'auth0')."""
        ...


@lru_cache()
def get_auth_provider() -> AuthProvider:
    """Factory — returns the configured auth provider singleton."""
    from josi.core.config import settings

    provider = getattr(settings, "auth_provider_name", "clerk")

    if provider == "clerk":
        from josi.auth.providers.clerk import ClerkProvider
        return ClerkProvider()
    else:
        raise ValueError(f"Unknown auth provider: {provider}")
```

"""Clerk auth provider adapter."""
import httpx
from fastapi import HTTPException, status

import structlog

from josi.auth.providers.base import AuthProvider
from josi.core.config import settings

logger = structlog.get_logger()


class ClerkProvider(AuthProvider):

    @property
    def provider_name(self) -> str:
        return "clerk"

    def validate_jwt(self, token: str) -> dict:
        import jwt as pyjwt
        from jwt import PyJWKClient

        try:
            jwks_url = "https://api.clerk.com/.well-known/jwks.json"
            jwks_client = PyJWKClient(jwks_url)
            signing_key = jwks_client.get_signing_key_from_jwt(token)
            claims = pyjwt.decode(
                token,
                signing_key.key,
                algorithms=["RS256"],
                options={"verify_aud": False},
            )
            return claims
        except Exception as e:
            logger.warning("Clerk JWT validation failed", error=str(e))
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid or expired session token",
                headers={"WWW-Authenticate": "Bearer"},
            )

    async def set_user_metadata(self, provider_user_id: str, metadata: dict) -> bool:
        try:
            async with httpx.AsyncClient() as client:
                response = await client.patch(
                    f"https://api.clerk.com/v1/users/{provider_user_id}",
                    headers={
                        "Authorization": f"Bearer {settings.clerk_secret_key}",
                        "Content-Type": "application/json",
                    },
                    json={"public_metadata": metadata},
                )
                if response.status_code != 200:
                    logger.error(
                        "Failed to set Clerk publicMetadata",
                        provider_user_id=provider_user_id,
                        status=response.status_code,
                        body=response.text,
                    )
                    return False
                return True
        except Exception as e:
            logger.error("Clerk API call failed", error=str(e))
            return False
```

```python
"""Redis-backed session cache for fast auth resolution.

Uses JSON serialization (not pickle) for security — CurrentUser is a
simple Pydantic model with only primitive types.
"""
import json
from typing import Optional
from uuid import UUID

import structlog

from josi.auth.schemas import CurrentUser
from josi.core.cache import cache_manager

logger = structlog.get_logger()

SESSION_TTL = 900  # 15 minutes


def _cache_key(auth_provider_id: str) -> str:
    return f"josi:user_session:{auth_provider_id}"


async def get_cached_user(auth_provider_id: str) -> Optional[CurrentUser]:
    """Look up cached CurrentUser by auth_provider_id. Returns None on miss."""
    key = _cache_key(auth_provider_id)
    try:
        if not cache_manager.redis_client:
            return None
        raw = await cache_manager.redis_client.get(key)
        if raw is None:
            return None
        data = json.loads(raw)
        return CurrentUser.model_validate(data)
    except Exception:
        logger.warning("Session cache read failed", auth_provider_id=auth_provider_id)
        return None


async def cache_user(user: CurrentUser) -> None:
    """Cache a CurrentUser keyed by auth_provider_id (JSON serialized)."""
    key = _cache_key(user.auth_provider_id)
    try:
        if not cache_manager.redis_client:
            return
        payload = json.dumps(user.model_dump(mode="json"))
        await cache_manager.redis_client.setex(key, SESSION_TTL, payload)
    except Exception:
        logger.warning("Session cache write failed", auth_provider_id=user.auth_provider_id)


async def invalidate_user(auth_provider_id: str) -> None:
    """Remove a user's session from cache (e.g., on logout, subscription change)."""
    await cache_manager.delete(_cache_key(auth_provider_id))
    logger.info("Session cache invalidated", auth_provider_id=auth_provider_id)


async def invalidate_user_by_id(user_id: UUID) -> None:
    """Remove session by user_id — scans for matching key.

    Use sparingly (for admin actions). Prefer invalidate_user() with provider_id.
    """
    await cache_manager.delete("josi:user_session:*")
    logger.info("Session cache bulk invalidated for user_id scan", user_id=str(user_id))
```

**Step 2: Add cache layer to middleware**

In `src/josi/auth/middleware.py`, update `resolve_current_user()` to check Redis first:

```python
from josi.services.session_cache_service import get_cached_user, cache_user
from josi.auth.providers import get_auth_provider

async def resolve_current_user(request: Request) -> CurrentUser:
    auth_header: Optional[str] = request.headers.get("authorization")
    api_key_header: Optional[str] = request.headers.get("x-api-key")

    # Path 1: JWT (B2C)
    if auth_header and auth_header.lower().startswith("bearer "):
        token = auth_header[7:]
        provider = get_auth_provider()
        claims = provider.validate_jwt(token)

        auth_provider_id = claims.get("sub", "")

        # Check Redis cache first
        cached = await get_cached_user(auth_provider_id)
        if cached:
            return cached

        # Cache miss — resolve from DB
        user_service = UserService()
        current_user = await user_service.resolve_user_from_db(
            auth_provider_id=auth_provider_id,
            email=claims.get("email", ""),
        )

        # Cache for next request
        await cache_user(current_user)
        return current_user

    # Path 2: API Key (B2B) — no caching (API keys are already fast)
    if api_key_header:
        api_key_service = ApiKeyService()
        result = await api_key_service.resolve_user_from_api_key(api_key_header)

        if not result:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid or expired API key",
            )

        user, api_key = result
        return CurrentUser(
            user_id=user.user_id,
            auth_provider_id=user.auth_provider_id,
            email=user.email,
            subscription_tier=user.subscription_tier_name or "Free",
            subscription_tier_id=user.subscription_tier_id,
            roles=user.roles,
            is_active=user.is_active,
            is_verified=user.is_verified,
        )

    raise HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Authentication required. Provide Authorization: Bearer <token> or X-API-Key header.",
    )
```

**Key change:** We no longer use `extract_user_from_claims()` (which read Clerk's custom JWT claims from publicMetadata). Instead we always go: Redis cache → DB fallback. This eliminates the dependency on Clerk's publicMetadata entirely.

"""Repository for UserUsage database operations."""
from typing import Optional
from datetime import datetime
from uuid import UUID

from sqlalchemy import select, and_

from josi.auth.schemas import CurrentUser
from josi.db.async_db import get_async_session
from josi.models.user_usage_model import UserUsage


def _current_period() -> str:
    return datetime.utcnow().strftime("%Y-%m")


class UsageRepository:

    def __init__(self, current_user: Optional[CurrentUser] = None):
        self.current_user = current_user

    async def get_or_create(self, user_id: UUID, period: Optional[str] = None) -> UserUsage:
        period = period or _current_period()
        async with get_async_session() as session:
            result = await session.execute(
                select(UserUsage).where(
                    and_(UserUsage.user_id == user_id, UserUsage.period == period)
                )
            )
            usage = result.scalar_one_or_none()
            if usage:
                return usage

            usage = UserUsage(user_id=user_id, period=period)
            session.add(usage)
            await session.flush()
            await session.refresh(usage)
            return usage

    async def increment(self, user_id: UUID, field: str, amount: int = 1) -> UserUsage:
        usage = await self.get_or_create(user_id)
        current = getattr(usage, field, 0)
        setattr(usage, field, current + amount)
        usage.updated_at = datetime.utcnow()
        async with get_async_session() as session:
            merged = await session.merge(usage)
            await session.flush()
            await session.refresh(merged)
            return merged
```

from josi.repositories.user_repository import UserRepository

logger = structlog.get_logger()

USAGE_TTL = 86400  # 24 hours


def _current_period() -> str:
    return datetime.utcnow().strftime("%Y-%m")


def _redis_key(user_id: UUID, period: str, field: str) -> str:
    return f"josi:usage:{user_id}:{period}:{field}"


class UsageService:

    TRACKED_FIELDS = [
        "charts_calculated",
        "ai_interpretations_used",
        "consultations_booked",
        "saved_charts_count",
    ]

    def __init__(self, current_user: Optional[CurrentUser] = None):
        self.current_user = current_user
        self.usage_repo = UsageRepository(current_user=current_user)
        self.user_repo = UserRepository(current_user=current_user)

    async def increment(self, user_id: UUID, field: str, amount: int = 1) -> int:
        """Increment a usage counter. Uses Redis for speed, syncs to DB."""
        if field not in self.TRACKED_FIELDS:
            raise ValueError(f"Unknown usage field: {field}")

        period = _current_period()
        key = _redis_key(user_id, period, field)

        # Try Redis atomic increment
        try:
            if cache_manager.redis_client and hasattr(cache_manager.redis_client, "incrby"):
                new_val = await cache_manager.redis_client.incrby(key, amount)
                await cache_manager.redis_client.expire(key, USAGE_TTL)
                return new_val
        except Exception:
            pass

        # Fallback: direct DB increment
        usage = await self.usage_repo.increment(user_id, field, amount)
        return getattr(usage, field, 0)

    async def get_usage(self, user_id: UUID, period: Optional[str] = None) -> UserUsageResponse:
        """Get current usage for a user with tier limits."""
        period = period or _current_period()

        # Try Redis first for each field
        usage_data = {}
        for field in self.TRACKED_FIELDS:
            key = _redis_key(user_id, period, field)
            try:
                if cache_manager.redis_client:
                    val = await cache_manager.redis_client.get(key)
                    if val is not None:
                        usage_data[field] = int(val)
                        continue
            except Exception:
                pass
            usage_data[field] = None  # Will fall back to DB

        # If any field missed Redis, fetch from DB
        if any(v is None for v in usage_data.values()):
            db_usage = await self.usage_repo.get_or_create(user_id, period)
            for field in self.TRACKED_FIELDS:
                if usage_data[field] is None:
                    usage_data[field] = getattr(db_usage, field, 0)

        # Get tier limits
        user = await self.user_repo.get_by_id(user_id)
        limits = user.get_tier_limits() if user else {}

        return UserUsageResponse(
            period=period,
            charts_calculated=usage_data.get("charts_calculated", 0),
            ai_interpretations_used=usage_data.get("ai_interpretations_used", 0),
            consultations_booked=usage_data.get("consultations_booked", 0),
            saved_charts_count=usage_data.get("saved_charts_count", 0),
            limits=limits,
        )

    async def sync_to_db(self, user_id: UUID, period: Optional[str] = None) -> None:
        """Flush Redis counters to DB. Call periodically or on usage check."""
        period = period or _current_period()
        usage = await self.usage_repo.get_or_create(user_id, period)

        for field in self.TRACKED_FIELDS:
            key = _redis_key(user_id, period, field)
            try:
                if cache_manager.redis_client:
                    val = await cache_manager.redis_client.get(key)
                    if val is not None:
                        setattr(usage, field, int(val))
            except Exception:
                continue

        from josi.db.async_db import get_async_session
        async with get_async_session() as session:
            await session.merge(usage)
```

    update_data = body.model_dump(exclude_unset=True)

    return user

```typescript
'use client';

import { createContext, useContext, useEffect, useState, useCallback, useRef } from 'react';
import { useAuth as useClerkAuth, useUser as useClerkUser } from '@clerk/nextjs';
import { setAsyncTokenGetter } from '@/lib/api-client';
import { setAsyncGraphQLTokenGetter } from '@/lib/graphql-client';
import { apiClient } from '@/lib/api-client';

interface UserProfile {
  user_id: string;
  email: string;
  full_name: string;
  phone?: string;
  avatar_url?: string;
  subscription_tier_id?: number;
  subscription_tier_name?: string;
  roles: string[];
  is_active: boolean;
  is_verified: boolean;
}

interface AuthContextType {
  user: {
    id: string;
    auth_provider_id: string;
    email: string;
    name: string;
    phone?: string;
    subscription_tier: string;
    roles: string[];
  } | null;
  isLoading: boolean;
  isAuthenticated: boolean;
  getToken: () => Promise<string | null>;
  logout: () => Promise<void>;
  refreshUser: () => Promise<void>;
}

const AuthContext = createContext<AuthContextType>({
  user: null,
  isLoading: true,
  isAuthenticated: false,
  getToken: async () => null,
  logout: async () => {},
  refreshUser: async () => {},
});

export function AuthContextProvider({ children }: { children: React.ReactNode }) {
  const { isLoaded, isSignedIn, getToken, signOut } = useClerkAuth();
  const { user: clerkUser, isLoaded: isUserLoaded } = useClerkUser();
  const [userProfile, setUserProfile] = useState<UserProfile | null>(null);
  const [isLoadingProfile, setIsLoadingProfile] = useState(false);
  const fetchedRef = useRef(false);

  // Wire the async token getter into the API and GraphQL clients
  useEffect(() => {
    setAsyncTokenGetter(getToken);
    setAsyncGraphQLTokenGetter(getToken);
  }, [getToken]);

  // Fetch user profile from our API (source of truth — not Clerk metadata)
  const fetchProfile = useCallback(async () => {
    if (!isSignedIn || fetchedRef.current) return;
    fetchedRef.current = true;
    setIsLoadingProfile(true);

    try {
      const response = await apiClient.get<UserProfile>('/me');
      if (response.success && response.data) {
        setUserProfile(response.data);
      }
    } catch (err) {
      console.warn('Failed to fetch user profile:', err);
    } finally {
      setIsLoadingProfile(false);
    }
  }, [isSignedIn]);

  useEffect(() => {
    fetchProfile();
  }, [fetchProfile]);

  // Reset on sign-out
  useEffect(() => {
    if (!isSignedIn) {
      fetchedRef.current = false;
      setUserProfile(null);
    }
  }, [isSignedIn]);

  const refreshUser = useCallback(async () => {
    fetchedRef.current = false;
    await fetchProfile();
  }, [fetchProfile]);

  const user = userProfile
    ? {
        id: userProfile.user_id,
        auth_provider_id: clerkUser?.id || '',
        email: userProfile.email,
        name: userProfile.full_name,
        phone: userProfile.phone,
        subscription_tier: userProfile.subscription_tier_name || 'Free',
        roles: userProfile.roles || ['user'],
      }
    : null;

  const logout = async () => {
    // Invalidate server-side session cache
    try {
      const token = await getToken();
      if (token) {
        await fetch(
          `${process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000'}/api/v1/auth/logout`,
          {
            method: 'POST',
            headers: { Authorization: `Bearer ${token}` },
          }
        );
      }
    } catch {
      // Non-fatal
    }
    await signOut();
    window.location.href = '/auth/login';
  };

  return (
    <AuthContext.Provider
      value={{
        user,
        isLoading: !isLoaded || !isUserLoaded || isLoadingProfile,
        isAuthenticated: !!isSignedIn,
        getToken,
        logout,
        refreshUser,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => useContext(AuthContext);
```

**Key changes:**
- User data comes from `GET /api/v1/me` (our DB), not Clerk's `publicMetadata`
- `refreshUser()` exposed for re-fetching after subscription changes
- Logout calls `POST /api/v1/auth/logout` to invalidate Redis cache
- No more sync-claims logic — the server handles everything
